141void* sbl_arena_alloc(SblArena* arena, uint64_t size) {
142 if (!arena || !arena->current) return NULL;
143 uint64_t align = 8;
144 SblArenaBlock* b = arena->current;
145
146 uintptr_t curr_ptr = (uintptr_t)b + sizeof(SblArenaBlock) + b->offset;
147 uintptr_t aligned_ptr = sbl_arena__align_forward(curr_ptr, align);
148 uint64_t aligned_offset = aligned_ptr - ((uintptr_t)b + sizeof(SblArenaBlock));
149
150 if (aligned_offset + size > b->size) {
151 uint64_t next_size = b->size * 2;
152 if (size > next_size)
153 next_size = size * 2;
154 SblArenaBlock* next = sbl_arena__block_create(next_size);
155 if (!next) return NULL;
156 b->next = next;
157 arena->current = next;
158 b = next;
159 aligned_ptr = (uintptr_t)b + sizeof(SblArenaBlock);
160 }
161
162 void* ptr = (void*)aligned_ptr;
163 b->offset = (aligned_ptr + size) - ((uintptr_t)b + sizeof(SblArenaBlock));
164 return ptr;
165}
