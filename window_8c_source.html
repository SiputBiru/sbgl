7static LRESULT CALLBACK WindowProc(HWND hwnd, UINT msg, WPARAM wparam, LPARAM lparam) {
8 sbgl_Window* window = (sbgl_Window*)GetWindowLongPtr(hwnd, GWLP_USERDATA);

34 if (window)
35 window->focused = false;
36 break;
37 }
38
39 // Pass input messages to the input system
40 if (window)
41 win32_internal_process_message(window->input, msg, wparam, lparam);
42
43 return DefWindowProc(hwnd, msg, wparam, lparam);

144void sbgl_os_SetCursorLocked(sbgl_Window* window, bool locked) {
145 if (!window)
146 return;
147
148 if (locked) {
149 /* The cursor is confined to the window client area by clipping it
150 to the screen-space coordinates of the client rectangle. */
151 RECT rect;
152 GetClientRect(window->hwnd, &rect);
153 ClientToScreen(window->hwnd, (POINT*)&rect.left);
154 ClientToScreen(window->hwnd, (POINT*)&rect.right);
155 ClipCursor(&rect);
156
157 /* The cursor is initially centered within the window to ensure
158 consistent relative motion tracking. */
159 int centerX = window->width / 2;
160 int centerY = window->height / 2;
161 POINT p = { centerX, centerY };
162 ClientToScreen(window->hwnd, &p);
163 SetCursorPos(p.x, p.y);
164 } else {
165 /* Any existing cursor clipping is removed, allowing the pointer
166 to move freely across the entire desktop. */
167 ClipCursor(NULL);
168 }
169}
