157static inline void sbgl_RaySphereIntersectBatch(
158 sbgl_Ray ray,
159 const sbgl_Sphere* spheres,
160 sbgl_HitResult* results,
161 uint32_t count
162) {
163 for (uint32_t i = 0; i < count; ++i) {
164 sbgl_Vec3 oc = sbgl_Vec3Sub(ray.origin, spheres[i].center);
165 float a = sbgl_Vec3Dot(ray.direction, ray.direction);
166 float b = 2.0f * sbgl_Vec3Dot(oc, ray.direction);
167 float c = sbgl_Vec3Dot(oc, oc) - spheres[i].radius * spheres[i].radius;
168 float discriminant = b * b - 4 * a * c;
169
170 results[i].hit = false;
171 if (discriminant >= 0) {
172 float sqrt_d = sqrtf(discriminant);
173 float t0 = (-b - sqrt_d) / (2.0f * a);
174 float t1 = (-b + sqrt_d) / (2.0f * a);
175
176 float t = t0;
177 if (t < 0)
178 t = t1;
179
180 if (t > 0) {
181 results[i].hit = true;
182 results[i].distance = t;
183 results[i].point = sbgl_Vec3Add(ray.origin, sbgl_Vec3Mul(ray.direction, t));
184 results[i].normal =
185 sbgl_Vec3Normalize(sbgl_Vec3Sub(results[i].point, spheres[i].center));
186 }
187 }
188 }
189}

197static inline void sbgl_RayAABBIntersectBatch(
198 sbgl_Ray ray,
199 const sbgl_AABB* boxes,
200 sbgl_HitResult* results,
201 uint32_t count
202) {
203 sbgl_Vec3 inv_dir = {
204 { 1.0f / ray.direction.x, 1.0f / ray.direction.y, 1.0f / ray.direction.z, 0.0f }
205 };
206
207 for (uint32_t i = 0; i < count; ++i) {
208 float t1 = (boxes[i].min.x - ray.origin.x) * inv_dir.x;
209 float t2 = (boxes[i].max.x - ray.origin.x) * inv_dir.x;
210 float tmin = fminf(t1, t2);
211 float tmax = fmaxf(t1, t2);
212
213 t1 = (boxes[i].min.y - ray.origin.y) * inv_dir.y;
214 t2 = (boxes[i].max.y - ray.origin.y) * inv_dir.y;
215 tmin = fmaxf(tmin, fminf(t1, t2));
216 tmax = fminf(tmax, fmaxf(t1, t2));
217
218 t1 = (boxes[i].min.z - ray.origin.z) * inv_dir.z;
219 t2 = (boxes[i].max.z - ray.origin.z) * inv_dir.z;
220 tmin = fmaxf(tmin, fminf(t1, t2));
221 tmax = fminf(tmax, fmaxf(t1, t2));
222
223 results[i].hit = (tmax >= tmin && tmax > 0);
224 if (results[i].hit) {
225 results[i].distance = tmin > 0 ? tmin : tmax;
226 results[i].point =
227 sbgl_Vec3Add(ray.origin, sbgl_Vec3Mul(ray.direction, results[i].distance));
228
229 // Calculate normal based on which face was hit
230 sbgl_Vec3 center = sbgl_Vec3Mul(sbgl_Vec3Add(boxes[i].min, boxes[i].max), 0.5f);
231 sbgl_Vec3 p = sbgl_Vec3Sub(results[i].point, center);
232 sbgl_Vec3 d = sbgl_Vec3Mul(sbgl_Vec3Sub(boxes[i].max, boxes[i].min), 0.5f);
233 float epsilon = 0.0001f;
234
235 results[i].normal = sbgl_Vec3Set(0, 0, 0);
236 if (fabsf(p.x) >= fabsf(d.x) - epsilon)
237 results[i].normal.x = p.x > 0 ? 1.0f : -1.0f;
238 else if (fabsf(p.y) >= fabsf(d.y) - epsilon)
239 results[i].normal.y = p.y > 0 ? 1.0f : -1.0f;
240 else if (fabsf(p.z) >= fabsf(d.z) - epsilon)
241 results[i].normal.z = p.z > 0 ? 1.0f : -1.0f;
242 }
243 }
