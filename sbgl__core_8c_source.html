109 inner->clearColor[0] = 0.0f;

125 }

128 if (!inner->gfx) {
129 res.error = SBGL_ERROR_GRAPHICS_INITIALIZATION_FAILED;
130 sbgl_os_DestroyWindow(inner->window);
131 sbl_arena_free(&inner->arena);

133 }

141 return;
142 sbgl_InternalContext* inner = (sbgl_InternalContext*)ctx->inner;

350sbgl_LoadShaderFromFile(sbgl_Context* ctx, sbgl_ShaderStage stage, const char* filename) {
351 FILE* file = fopen(filename, "rb");
352 if (!file) {
353 char fallback[256];
354 // Try examples/shaders/ prefix
355 snprintf(fallback, sizeof(fallback), "examples/%s", filename);
356 file = fopen(fallback, "rb");
357 if (!file) {
358 // Try build/examples/ prefix
359 snprintf(fallback, sizeof(fallback), "build/examples/%s", filename);
360 file = fopen(fallback, "rb");
361 if (!file) {
362 sbgl_internal_log(SBGL_LOG_ERROR, "Failed to open shader file.");
363 return SBGL_INVALID_HANDLE;
364 }
365 }
366 }
367
368 fseek(file, 0, SEEK_END);
369 size_t size = ftell(file);
370 fseek(file, 0, SEEK_SET);
371
372 uint32_t* buffer = malloc(size);
373 if (!buffer) {
374 fclose(file);
375 return SBGL_INVALID_HANDLE;
376 }
377
378 size_t read = fread(buffer, 1, size, file);
379 fclose(file);
380
381 if (read != size) {
382 free(buffer);
383 return SBGL_INVALID_HANDLE;
384 }
385
386 sbgl_Shader shader = sbgl_LoadShader(ctx, stage, buffer, size);
387 free(buffer);
388 return shader;
389}

499sbgl_RenderQueue* sbgl_CreateRenderQueue(sbgl_Context* ctx, SblArena* arena) {
500 if (!ctx || !arena) {
501 return NULL;
502 }
503
504 // Allocate the queue structure from the provided arena
505 sbgl_RenderQueue* queue = SBL_ARENA_PUSH_STRUCT_ZERO(arena, sbgl_RenderQueue);
506 if (!queue) {
507 ctx->result = SBGL_ERROR_OUT_OF_MEMORY;
508 return NULL;
509 }
510
511 // Initialize the queue with a default capacity of 16,384 packets
512 /* Initialize the queue with a high capacity to support massive batching (e.g., voxels). */
513 queue->capacity = 131072;
514 queue->count = 0;
515 queue->arena = arena;
516 queue->packets = SBL_ARENA_PUSH_ARRAY_ZERO(arena, sbgl_DrawPacket, queue->capacity);
517 queue->instances = SBL_ARENA_PUSH_ARRAY_ZERO(arena, sbgl_InstanceData, queue->capacity);
518
519 if (!queue->packets || !queue->instances) {
520 ctx->result = SBGL_ERROR_OUT_OF_MEMORY;
521 return NULL;
522 }
523
524 ctx->result = SBGL_SUCCESS;
525 return queue;
526}

528void sbgl_SubmitDraw(
529 sbgl_RenderQueue* queue,
530 uint32_t mesh,
531 uint32_t material,
532 uint32_t blendMode,
533 uint32_t sidedness,
534 uint32_t tags,
535 sbgl_SortKey key,
536 const sbgl_InstanceData* data
537) {
538 if (!queue) {
539 return;
540 }
541
542 // Verify the queue has not reached maximum capacity
543 if (queue->count >= queue->capacity) {
544 return;
545 }
546
547 // The system appends the draw packet to the contiguous array
548 uint32_t index = queue->count++;
549 sbgl_DrawPacket* packet = &queue->packets[index];
550 packet->key = key;
551 packet->header = SBGL_PACK_HEADER(mesh, material, blendMode, sidedness, tags);
552
553 // Per-instance data is copied into the parallel array
554 if (data) {
555 queue->instances[index] = *data;
556 } else {
557 // An identity transform and white color are used if no data is provided
558 memset(&queue->instances[index], 0, sizeof(sbgl_InstanceData));
559 queue->instances[index].transform = sbgl_Mat4Identity();
560 queue->instances[index].color = sbgl_Vec4Set(1, 1, 1, 1);
561 }
562}

573void sbgl_RenderQueuesEx(
574 sbgl_Context* ctx,
575 sbgl_RenderQueue** queues,
576 uint32_t queueCount,
577 const sbgl_Mat4* viewProj,
578 uint64_t userAddress
579) {
580 if (!ctx || !ctx->inner || !queues || queueCount == 0) {
581 return;
582 }
583
584 sbgl_InternalContext* inner = (sbgl_InternalContext*)ctx->inner;
585
586 // Total packet count is calculated across all queues to determine buffer requirements
587 uint32_t totalPackets = 0;
588 for (uint32_t i = 0; i < queueCount; ++i) {
589 if (queues[i]) {
590 totalPackets += queues[i]->count;
591 }
592 }
593
594 if (totalPackets == 0) {
595 return;
596 }
597
598 // The transient arena is reset for this frame's batching work
599 sbl_arena_reset(&inner->transientArena);
600
601#ifdef linux
602 inner->sortStartTime = sbgl_internal_GetTime();
603#endif
604
605 // Temporary host memory is allocated for merging and sorting from the transient arena
606 sbgl_DrawPacket* mergedPackets =
607 SBL_ARENA_PUSH_ARRAY(&inner->transientArena, sbgl_DrawPacket, totalPackets);
608 sbgl_InstanceData* mergedInstances =
609 SBL_ARENA_PUSH_ARRAY(&inner->transientArena, sbgl_InstanceData, totalPackets);
610 sbgl_SortKey* keys = SBL_ARENA_PUSH_ARRAY(&inner->transientArena, sbgl_SortKey, totalPackets);
611 uint32_t* indices = SBL_ARENA_PUSH_ARRAY(&inner->transientArena, uint32_t, totalPackets);
612
613 if (!mergedPackets || !mergedInstances || !keys || !indices) {
614 ctx->result = SBGL_ERROR_OUT_OF_MEMORY;
615 return;
616 }
617
618 // Packets and instance data are merged from all queues
619 uint32_t current = 0;
620 for (uint32_t i = 0; i < queueCount; ++i) {
621 if (!queues[i])
622 continue;
623 for (uint32_t j = 0; j < queues[i]->count; ++j) {
624 mergedPackets[current] = queues[i]->packets[j];
625 mergedInstances[current] = queues[i]->instances[j];
626 keys[current] = mergedPackets[current].key;
627 indices[current] = current;
628 current++;
629 }
630 // The queue is cleared for the next frame
631 queues[i]->count = 0;
632 }
633
634 // A stable radix sort is performed on the packets based on sort keys
635 sbgl_SortKey* temp_keys =
636 SBL_ARENA_PUSH_ARRAY(&inner->transientArena, sbgl_SortKey, totalPackets);
637 uint32_t* temp_indices = SBL_ARENA_PUSH_ARRAY(&inner->transientArena, uint32_t, totalPackets);
638 if (!temp_keys || !temp_indices) {
639 ctx->result = SBGL_ERROR_OUT_OF_MEMORY;
640 return;
641 }
642
643 sbgl_radix_sort(keys, indices, totalPackets, temp_keys, temp_indices);
644
645 // Packets and instances are reordered according to the sorted indices
646 sbgl_DrawPacket* sortedPackets =
647 SBL_ARENA_PUSH_ARRAY(&inner->transientArena, sbgl_DrawPacket, totalPackets);
648 sbgl_InstanceData* sortedInstances =
649 SBL_ARENA_PUSH_ARRAY(&inner->transientArena, sbgl_InstanceData, totalPackets);
650 if (!sortedPackets || !sortedInstances) {
651 ctx->result = SBGL_ERROR_OUT_OF_MEMORY;
652 return;
653 }
654
655 for (uint32_t i = 0; i < totalPackets; ++i) {
656 sortedPackets[i] = mergedPackets[indices[i]];
657 sortedInstances[i] = mergedInstances[indices[i]];
658 }
659
660 // Sorted packets are baked into optimized indirect draw commands
661 sbgl_IndirectCommand* commands =
662 SBL_ARENA_PUSH_ARRAY(&inner->transientArena, sbgl_IndirectCommand, totalPackets);
663 if (!commands) {
664 ctx->result = SBGL_ERROR_OUT_OF_MEMORY;
665 return;
666 }
667
668 uint32_t commandCount = sbgl_bake_commands(sortedPackets, totalPackets, commands, totalPackets);
669
670#ifdef linux
671 struct timespec sortEndTime = sbgl_internal_GetTime();
672 inner->currentFrame.cpu_sort_time +=
673 sbgl_internal_GetElapsedMs(inner->sortStartTime, sortEndTime);
674#endif
675
676 inner->currentFrame.draw_calls += commandCount;
677 inner->currentFrame.instance_count += totalPackets;
678
679 if (commandCount > 0) {
680 // Allocate transient GPU space for packed instance data
681 sbgl_GfxTransientAllocation instanceAlloc = sbgl_gfx_AllocateTransient(
682 inner->gfx,
683 sizeof(sbgl_InstanceData) * totalPackets,
684 256 // Storage buffer alignment
685 );
686
687 if (instanceAlloc.buffer != SBGL_INVALID_HANDLE) {
688 memcpy(instanceAlloc.mapped, sortedInstances, sizeof(sbgl_InstanceData) * totalPackets);
689
690 // Retrieve BDA address and update push constants
691 uint64_t bdaAddress = instanceAlloc.deviceAddress;
692
693 struct {
694 sbgl_Mat4 viewProj;
695 uint64_t instanceAddress;
696 uint64_t userAddress;
697 } pc;
698 pc.viewProj = viewProj ? *viewProj : sbgl_Mat4Identity();
699 pc.instanceAddress = bdaAddress;
700 pc.userAddress = userAddress;
701
702 sbgl_gfx_PushConstants(inner->gfx, sizeof(pc), &pc);
703
704 // Allocate transient GPU space for baked commands and dispatch MDI
705 sbgl_GfxTransientAllocation indirectAlloc = sbgl_gfx_AllocateTransient(
706 inner->gfx,
707 sizeof(sbgl_IndirectCommand) * commandCount,
