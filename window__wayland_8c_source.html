104void sbgl_os_PollEvents(sbgl_Window* window) {
105 if (!window) return;
106
107 // Clear deltas if relative pointer is active to allow accumulation in listeners
108 if (window->relative_pointer && window->input) {
109 window->input->mouseDeltaX = 0;
110 window->input->mouseDeltaY = 0;
111 }
112
113 /* The system ensures that pending events are dispatched and new events are read
114 from the display in a non-blocking manner to prevent main loop stalls. */
115 while (wl_display_prepare_read(window->display) != 0) {
116 wl_display_dispatch_pending(window->display);
117 }
118 wl_display_flush(window->display);
119
120 struct pollfd pfd = { .fd = wl_display_get_fd(window->display), .events = POLLIN };
121 if (poll(&pfd, 1, 0) > 0) {
122 wl_display_read_events(window->display);
123 } else {
124 wl_display_cancel_read(window->display);
125 }
126
127 wl_display_dispatch_pending(window->display);
