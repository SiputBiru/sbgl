77 * Create a simple index buffer for the unit cube.

109 /* Ensure event polling and GPU command buffer start happen before input query */

141 if (input->keysDown[SBGL_KEY_S])
142 camera.position = sbgl_Vec3Sub(camera.position, sbgl_Vec3Mul(front, velocity));
143 if (input->keysDown[SBGL_KEY_A])
144 camera.position = sbgl_Vec3Sub(camera.position, sbgl_Vec3Mul(right, velocity));
145 if (input->keysDown[SBGL_KEY_D])
146 camera.position = sbgl_Vec3Add(camera.position, sbgl_Vec3Mul(right, velocity));
147 if (input->keysDown[SBGL_KEY_Q])
148 camera.position.y -= velocity;
149 if (input->keysDown[SBGL_KEY_E])
150 camera.position.y += velocity;
151
152 camera.target = sbgl_Vec3Add(camera.position, front);
153 camera.up = sbgl_Vec3Normalize(sbgl_Vec3Cross(right, front));

155 int width, height;

157 camera.aspect = (float)width / (float)height;
158

161

163 * Perform the voxel system update.
164 * This dispatches compute shaders only if the camera crosses a chunk boundary.
165 */
166 sbgl_Voxel_Update(voxelSys, camera.position);
167
168 /*
169 * Perform GPU-driven culling before starting the graphics pass.
170 */
171 sbgl_Voxel_Cull(voxelSys, viewProj);
172
173 /*
174 * Clear the framebuffer and begin the graphics pass.
175 */
176 sbgl_Clear(ctx, 0.5f, 0.7f, 1.0f, 1.0f);
177 sbgl_BeginDrawing(ctx);
178
179 /* Bind the rendering pipeline and geometry */
180 sbgl_BindPipeline(ctx, renderPipe);
181 sbgl_BindBuffer(ctx, iBuf, SBGL_BUFFER_USAGE_INDEX);
182
183 /* Setup graphics push constants using addresses retrieved from the voxel system */
184 RenderPushConstants rpc = { .viewProj = viewProj,
185 .aabbAddress = sbgl_Voxel_GetAABBAddress(voxelSys),
186 .voxelDataAddress = sbgl_Voxel_GetInstanceAddress(voxelSys),
187 .paletteAddress = sbgl_Voxel_GetPaletteAddress(voxelSys) };
188 sbgl_PushConstants(ctx, sizeof(rpc), &rpc);
189
190 /* Dispatch the voxel rendering commands (Graphics pass only) */
191 sbgl_Voxel_Render(voxelSys);
192
193 sbgl_EndDrawing(ctx);
194 }
195
196 /*
197 * Gracefully release all resources before shutting down.
198 */
199 sbgl_DeviceWaitIdle(ctx);
200 sbgl_Voxel_Destroy(voxelSys);
201 sbgl_DestroyPipeline(ctx, renderPipe);
202 sbgl_DestroyShader(ctx, vertShader);
203 sbgl_DestroyShader(ctx, fragShader);
204 sbgl_DestroyBuffer(ctx, iBuf);
205 sbgl_Shutdown(ctx);
206
207 return 0;
