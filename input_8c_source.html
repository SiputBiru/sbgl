4static SBGL_Scancode win32_vk_to_scancode(WPARAM wparam) {
5 if (wparam >= 'A' && wparam <= 'Z') return (SBGL_Scancode)(SBGL_SCANCODE_A + (wparam - 'A'));
6 if (wparam == '0') return SBGL_SCANCODE_0;
7 if (wparam >= '1' && wparam <= '9') return (SBGL_Scancode)(SBGL_SCANCODE_1 + (wparam - '1'));
8
9 switch (wparam) {
10 case VK_ESCAPE: return SBGL_SCANCODE_ESCAPE;
11 case VK_RETURN: return SBGL_SCANCODE_RETURN;
12 case VK_BACK: return SBGL_SCANCODE_BACKSPACE;
13 case VK_TAB: return SBGL_SCANCODE_TAB;
14 case VK_SPACE: return SBGL_SCANCODE_SPACE;
15 case VK_UP: return SBGL_SCANCODE_UP;
16 case VK_DOWN: return SBGL_SCANCODE_DOWN;
17 case VK_LEFT: return SBGL_SCANCODE_LEFT;
18 case VK_RIGHT: return SBGL_SCANCODE_RIGHT;
19 case VK_SHIFT: return SBGL_SCANCODE_LSHIFT;
20 case VK_CONTROL: return SBGL_SCANCODE_LCTRL;
21 case VK_MENU: return SBGL_SCANCODE_LALT;
22 case 0xBB: return SBGL_SCANCODE_EQUAL; // VK_OEM_PLUS
23 case 0xBD: return SBGL_SCANCODE_MINUS; // VK_OEM_MINUS

25 }
