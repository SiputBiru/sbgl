25int32_t VoxelPool_AcquireSlot(VoxelPool* pool, sbgl_ivec3 pos, bool* is_new) {
26 int32_t lru_index = 0;
27 uint64_t min_frame = UINT64_MAX;
28
29 // Search for an existing slot with matching coordinates or find an empty slot.
30 for (uint32_t i = 0; i < pool->capacity; ++i) {
31 if (pool->active[i]) {
32 if (pool->positions[i].x == pos.x &&
33 pool->positions[i].y == pos.y &&
34 pool->positions[i].z == pos.z) {
35 // Found an existing slot, update its frame for LRU.
36 pool->last_used_frames[i] = pool->current_frame;
37 if (is_new) *is_new = false;
38 return (int32_t)i;
39 }
40 } else {
41 // Found an inactive slot, assign it immediately.
42 pool->active[i] = 1;
43 pool->positions[i] = pos;
44 pool->last_used_frames[i] = pool->current_frame;
45 if (is_new) *is_new = true;
46 return (int32_t)i;
47 }
48
49 // Keep track of the oldest used slot in case we need to recycle.
50 if (pool->last_used_frames[i] < min_frame) {
51 min_frame = pool->last_used_frames[i];
52 lru_index = (int32_t)i;
53 }
54 }
55
56 // No existing or inactive slots available; recycle the least recently used one.
57 pool->positions[lru_index] = pos;
58 pool->last_used_frames[lru_index] = pool->current_frame;
