221static uint32_t static_heap_alloc(sbgl_GfxContext* ctx, size_t size) {
222 /* The size is rounded up to a 256-byte boundary to satisfy the most stringent
223 Vulkan hardware alignment requirements (minUniformBufferOffsetAlignment). */
224 uint32_t alignedSize = (uint32_t)((size + 255) & ~255);
225
226 if (ctx->staticHeap.offset + alignedSize > ctx->staticHeap.size) {
227 /* If the requested allocation exceeds the remaining capacity of the static heap,
228 the context's result state is updated to reflect an out-of-memory error. */
229 ctx->result = SBGL_ERROR_OUT_OF_MEMORY;
230 return SBGL_INVALID_OFFSET;
231 }
232
233 uint32_t offset = ctx->staticHeap.offset;
234 ctx->staticHeap.offset += alignedSize;
235 return offset;
236}

238static uint32_t dynamic_heap_alloc(sbgl_GfxContext* ctx, size_t size) {
239 /* Dynamic allocations are performed within the heap slice corresponding to the
240 currently active frame, utilizing a 256-byte alignment for hardware compatibility. */
241 uint32_t alignedSize = (uint32_t)((size + 255) & ~255);
242 uint32_t frame = ctx->currentFrame;
243 uint32_t halfSize = ctx->dynamicHeap.size / 2;
244
245 if (ctx->dynamicHeap.offset[frame] + alignedSize > halfSize) {
246 /* If the dynamic allocation exceeds the current frame's capacity, the context
247 is marked with an out-of-memory error to prevent invalid GPU access. */
248 ctx->result = SBGL_ERROR_OUT_OF_MEMORY;
249 return SBGL_INVALID_OFFSET;
250 }
251
252 uint32_t offset = ctx->dynamicHeap.offset[frame];
253 ctx->dynamicHeap.offset[frame] += alignedSize;
254 return offset;
255}

257static uint32_t managed_heap_alloc(sbgl_GfxContext* ctx, size_t size) {
258 /* The sub-allocator utilizes a first-fit strategy across an array of memory
259 ranges, ensuring each allocation is aligned to a 256-byte boundary. */
260 uint32_t alignedSize = (uint32_t)((size + 255) & ~255);
261
262 for (uint32_t i = 0; i < ctx->managedHeap.rangeCount; i++) {
263 sbgl_GfxMemoryRange* range = &ctx->managedHeap.ranges[i];
264
265 /* A range is considered a candidate if it is currently unassigned (handle_index 0)
266 and possesses sufficient capacity to accommodate the requested aligned size. */
267 if (range->handle_index == 0 && range->size >= alignedSize) {
268 uint32_t offset = range->offset;
269
270 if (range->size > alignedSize) {
271 /* If the selected range is larger than the requested size, it is split into
272 two segments: the allocated block and a new unassigned trailing range. */
273 if (ctx->managedHeap.rangeCount >= 1024) {
274 /* The allocation fails if the maximum number of range descriptors is exceeded. */
275 ctx->result = SBGL_ERROR_OUT_OF_MEMORY;
276 return SBGL_INVALID_OFFSET;
277 }
278
279 /* Shift subsequent ranges to maintain the sequential order of the range array. */
280 for (uint32_t j = ctx->managedHeap.rangeCount; j > i + 1; j--) {
281 ctx->managedHeap.ranges[j] = ctx->managedHeap.ranges[j - 1];
282 }
283
284 /* Initialize the new free range representing the remaining space in the block. */
285 ctx->managedHeap.ranges[i + 1] = (sbgl_GfxMemoryRange){
286 .offset = offset + alignedSize,
287 .size = range->size - alignedSize,
288 .handle_index = 0
289 };
290
291 ctx->managedHeap.rangeCount++;
292 range->size = alignedSize;
293 }
294
295 /* The range is marked as active by assigning a non-zero handle index (1 is used
296 as a generic active marker in this implementation stage). */
297 range->handle_index = 1;
298 return offset;
299 }
300 }
301
302 /* If no suitable range is identified, the context state is updated to reflect
303 the allocation failure. */
304 ctx->result = SBGL_ERROR_OUT_OF_MEMORY;
305 return SBGL_INVALID_OFFSET;
306}

308static void managed_heap_free(sbgl_GfxContext* ctx, uint32_t offset) {
309 /* The freeing process identifies the range associated with the given offset
310 and attempts to merge it with adjacent free blocks to mitigate fragmentation. */
311 for (uint32_t i = 0; i < ctx->managedHeap.rangeCount; i++) {
312 if (ctx->managedHeap.ranges[i].offset == offset) {
313 /* The block is transitioned back to an unassigned state. */
314 ctx->managedHeap.ranges[i].handle_index = 0;
315
316 /* Coalescing with the subsequent block occurs if it is also unassigned. */
317 if (i + 1 < ctx->managedHeap.rangeCount && ctx->managedHeap.ranges[i + 1].handle_index == 0) {
318 ctx->managedHeap.ranges[i].size += ctx->managedHeap.ranges[i + 1].size;
319 for (uint32_t j = i + 1; j < ctx->managedHeap.rangeCount - 1; j++) {
320 ctx->managedHeap.ranges[j] = ctx->managedHeap.ranges[j + 1];
321 }
322 ctx->managedHeap.rangeCount--;
323 }
324
325 /* Coalescing with the preceding block occurs if it is also unassigned. */
326 if (i > 0 && ctx->managedHeap.ranges[i - 1].handle_index == 0) {
327 ctx->managedHeap.ranges[i - 1].size += ctx->managedHeap.ranges[i].size;
328 for (uint32_t j = i; j < ctx->managedHeap.rangeCount - 1; j++) {
329 ctx->managedHeap.ranges[j] = ctx->managedHeap.ranges[j + 1];
330 }
331 ctx->managedHeap.rangeCount--;
332 }
333
334 return;
335 }
336 }

339static bool create_heaps(sbgl_GfxContext* ctx) {
340 /* The system initializes three distinct memory heaps to support diverse resource
341 allocation patterns: Static for long-lived assets, Dynamic for per-frame data,
342 and Managed for flexible sub-allocations. All heaps are host-visible and
343 persistently mapped for zero-copy access. */
344
345 uint32_t memoryTypeIndex = find_memory_type(
346 ctx,
347 0xFFFFFFFF,
348 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT
349 );
350
351 if (memoryTypeIndex == SBGL_INVALID_OFFSET) {
352 return false;
353 }
354
355 VkMemoryAllocateFlagsInfo flagsInfo = {
356 .sType = VK_STRUCTURE_TYPE_MEMORY_ALLOCATE_FLAGS_INFO,
357 .flags = VK_MEMORY_ALLOCATE_DEVICE_ADDRESS_BIT,
358 };
359
360 VkMemoryAllocateInfo allocInfo = {
361 .sType = VK_STRUCTURE_TYPE_MEMORY_ALLOCATE_INFO,
362 .pNext = &flagsInfo,
363 .memoryTypeIndex = memoryTypeIndex,
364 };
365
366 // Static Heap
367 ctx->staticHeap.size = SBGL_STATIC_HEAP_SIZE;
368 allocInfo.allocationSize = ctx->staticHeap.size;
369 if (ctx->vk.vkAllocateMemory(ctx->device, &allocInfo, NULL, &ctx->staticHeap.memory) != VK_SUCCESS) {
370 return false;
371 }
372 ctx->vk.vkMapMemory(ctx->device, ctx->staticHeap.memory, 0, ctx->staticHeap.size, 0, &ctx->staticHeap.mapped);
373 ctx->staticHeap.offset = 0;
374
375 // Dynamic Heap (split into two 64MB buffers for double buffering)
376 ctx->dynamicHeap.size = SBGL_DYNAMIC_HEAP_SIZE;
377 allocInfo.allocationSize = ctx->dynamicHeap.size;
378 if (ctx->vk.vkAllocateMemory(ctx->device, &allocInfo, NULL, &ctx->dynamicHeap.memory) != VK_SUCCESS) {
379 return false;
380 }
381 void* dynamicBase;
382 ctx->vk.vkMapMemory(ctx->device, ctx->dynamicHeap.memory, 0, ctx->dynamicHeap.size, 0, &dynamicBase);
383 ctx->dynamicHeap.mapped[0] = dynamicBase;
384 ctx->dynamicHeap.mapped[1] = (char*)dynamicBase + (SBGL_DYNAMIC_HEAP_SIZE / 2);
385 ctx->dynamicHeap.offset[0] = 0;
386 ctx->dynamicHeap.offset[1] = 0;
387
388 // Managed Heap
389 ctx->managedHeap.size = SBGL_MANAGED_HEAP_SIZE;
390 allocInfo.allocationSize = ctx->managedHeap.size;
391 if (ctx->vk.vkAllocateMemory(ctx->device, &allocInfo, NULL, &ctx->managedHeap.memory) != VK_SUCCESS) {
392 return false;
393 }
394 ctx->vk.vkMapMemory(ctx->device, ctx->managedHeap.memory, 0, ctx->managedHeap.size, 0, &ctx->managedHeap.mapped);
395 ctx->managedHeap.rangeCount = 1;
396 ctx->managedHeap.ranges[0] = (sbgl_GfxMemoryRange){ .offset = 0, .size = SBGL_MANAGED_HEAP_SIZE, .handle_index = 0 };
397
398 (void)static_heap_alloc;
399 (void)dynamic_heap_alloc;
400 (void)managed_heap_alloc;
401 (void)managed_heap_free;
402
403 return true;
404}

682static bool create_swapchain(sbgl_GfxContext* ctx, sbgl_Window* window) {
683 int w, h;
684 sbgl_os_GetWindowSize(window, &w, &h);
685
686 VkSurfaceCapabilitiesKHR capabilities;
687 vkGetPhysicalDeviceSurfaceCapabilitiesKHR(ctx->physicalDevice, ctx->surface, &capabilities);
688
689 VkExtent2D extent = { (uint32_t)w, (uint32_t)h };
690 if (capabilities.currentExtent.width != 0xFFFFFFFF) {
691 extent = capabilities.currentExtent;
692 }
693
694 if (extent.width == 0 || extent.height == 0) {
695 return false;
696 }
697
698 uint32_t formatCount;
699 vkGetPhysicalDeviceSurfaceFormatsKHR(ctx->physicalDevice, ctx->surface, &formatCount, NULL);
700 if (formatCount == 0) {
701 fprintf(stderr, "[Vulkan] No supported surface formats found\n");
702 return false;
703 }
704 VkSurfaceFormatKHR formats[64];
705 if (formatCount > 64)
706 formatCount = 64;
707 vkGetPhysicalDeviceSurfaceFormatsKHR(ctx->physicalDevice, ctx->surface, &formatCount, formats);
708
709 VkSurfaceFormatKHR selectedFormat = formats[0];
710 for (uint32_t i = 0; i < formatCount; i++) {
711 if ((formats[i].format == VK_FORMAT_B8G8R8A8_SRGB ||
712 formats[i].format == VK_FORMAT_R8G8B8A8_SRGB) &&
713 formats[i].colorSpace == VK_COLOR_SPACE_SRGB_NONLINEAR_KHR) {
714 selectedFormat = formats[i];
715 break;
716 }
717 }
718
719 uint32_t imageCount = capabilities.minImageCount + 1;
720 if (capabilities.maxImageCount > 0 && imageCount > capabilities.maxImageCount) {
721 imageCount = capabilities.maxImageCount;
722 }
723
724 VkPresentModeKHR presentMode = VK_PRESENT_MODE_FIFO_KHR;
725 uint32_t presentModeCount;
726 vkGetPhysicalDeviceSurfacePresentModesKHR(
727 ctx->physicalDevice,
728 ctx->surface,
729 &presentModeCount,
730 NULL
731 );
732 if (presentModeCount > 0) {
733 VkPresentModeKHR presentModes[16];
734 if (presentModeCount > 16)
735 presentModeCount = 16;
736 vkGetPhysicalDeviceSurfacePresentModesKHR(
737 ctx->physicalDevice,
738 ctx->surface,
739 &presentModeCount,
740 presentModes
741 );
742
743 /* The system prioritizes present modes that minimize latency and maximize throughput.
744 IMMEDIATE is preferred for raw benchmarks, while MAILBOX provides high-performance
745 triple-buffering without tearing. */
746 bool mailboxSupported = false;
747 bool immediateSupported = false;
748 for (uint32_t i = 0; i < presentModeCount; i++) {
749 if (presentModes[i] == VK_PRESENT_MODE_IMMEDIATE_KHR)
750 immediateSupported = true;
751 if (presentModes[i] == VK_PRESENT_MODE_MAILBOX_KHR)
752 mailboxSupported = true;
753 }
754
755 if (immediateSupported) {
756 presentMode = VK_PRESENT_MODE_IMMEDIATE_KHR;
757 } else if (mailboxSupported) {
758 presentMode = VK_PRESENT_MODE_MAILBOX_KHR;
759 }
760 }
761
762 VkSwapchainCreateInfoKHR createInfo = {
763 .sType = VK_STRUCTURE_TYPE_SWAPCHAIN_CREATE_INFO_KHR,
764 .surface = ctx->surface,
765 .minImageCount = imageCount,
766 .imageFormat = selectedFormat.format,
767 .imageColorSpace = selectedFormat.colorSpace,
768 .imageExtent = extent,
769 .imageArrayLayers = 1,
770 .imageUsage = VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT,
771 .imageSharingMode = VK_SHARING_MODE_EXCLUSIVE,
772 .preTransform = capabilities.currentTransform,
773 .compositeAlpha = VK_COMPOSITE_ALPHA_OPAQUE_BIT_KHR,
774 .presentMode = presentMode,
775 .clipped = VK_TRUE,
776 };
777
778 if (ctx->vk.vkCreateSwapchainKHR(ctx->device, &createInfo, NULL, &ctx->swapchain) !=
779 VK_SUCCESS) {
780 fprintf(stderr, "[Vulkan] Failed to create swapchain\n");
781 return false;
782 }
783
784 ctx->swapchainExtent = createInfo.imageExtent;
785 ctx->swapchainFormat = createInfo.imageFormat;
786
787 ctx->vk.vkGetSwapchainImagesKHR(ctx->device, ctx->swapchain, &ctx->imageCount, NULL);
788
789 ctx->swapchainMark = sbl_arena_mark(ctx->arena);
790 ctx->images = SBL_ARENA_PUSH_ARRAY(ctx->arena, VkImage, ctx->imageCount);
791 if (!ctx->images)
792 return false;
793 ctx->vk.vkGetSwapchainImagesKHR(ctx->device, ctx->swapchain, &ctx->imageCount, ctx->images);
794
795 ctx->imageViews = SBL_ARENA_PUSH_ARRAY(ctx->arena, VkImageView, ctx->imageCount);
796 if (!ctx->imageViews)
797 return false;
798 for (uint32_t i = 0; i < ctx->imageCount; i++) {
799 VkImageViewCreateInfo viewInfo = {
800 .sType = VK_STRUCTURE_TYPE_IMAGE_VIEW_CREATE_INFO,
801 .image = ctx->images[i],
802 .viewType = VK_IMAGE_VIEW_TYPE_2D,
803 .format = ctx->swapchainFormat,
804 .subresourceRange = { .aspectMask = VK_IMAGE_ASPECT_COLOR_BIT,
805 .levelCount = 1,
806 .layerCount = 1 },
807 };
808 ctx->vk.vkCreateImageView(ctx->device, &viewInfo, NULL, &ctx->imageViews[i]);
809 }
810
811 printf(
812 "[Vulkan] Swapchain created (%dx%d, %u images, format: %d)\n",
813 ctx->swapchainExtent.width,
814 ctx->swapchainExtent.height,
815 ctx->imageCount,
816 ctx->swapchainFormat
817 );
818
819 if (!create_depth_resources(ctx))
820 return false;
821
822 return true;
823}

896static bool create_transient_resources(sbgl_GfxContext* ctx) {
897 /* The system allocates persistent, large-scale buffers for each frame in flight,
898 enabling high-frequency data updates without the overhead of per-frame allocations. */
899 for (uint32_t i = 0; i < SBGL_MAX_FRAMES_IN_FLIGHT; i++) {
900 ctx->transientBuffers[i] = sbgl_gfx_CreateBuffer(
901 ctx,
902 SBGL_BUFFER_USAGE_STORAGE | SBGL_BUFFER_USAGE_INDIRECT,
903 SBGL_TRANSIENT_BUFFER_SIZE,
904 NULL
905 );
906 if (ctx->transientBuffers[i] == SBGL_INVALID_HANDLE) {
907 return false;
908 }
909
910 uint32_t idx = (uint32_t)ctx->transientBuffers[i] - 1;
911 SBGL_VulkanBuffer* buffer = &ctx->buffers[idx];
912
913 /* The system utilizes the persistent mapping established during buffer creation
914 to provide zero-copy access to the transient memory pools. */
915 ctx->transientMapped[i] = buffer->mapped;
916 if (!ctx->transientMapped[i]) {
917 return false;
918 }
919 ctx->transientOffsets[i] = 0;
920 }
921 return true;
922}

952void sbgl_gfx_Shutdown(sbgl_GfxContext* ctx) {
953 if (!ctx)
954 return;
955
956 if (ctx->device) {
957 ctx->vk.vkDeviceWaitIdle(ctx->device);
958
959 // Clean up all active buffers
960 for (uint32_t i = 0; i < SBGL_MAX_BUFFERS; i++) {
961 if (ctx->bufferActive[i]) {
962 sbgl_gfx_DestroyBuffer(ctx, (sbgl_Buffer)(i + 1));
963 }
964 }
965
966 // Clean up all active shaders
967 for (uint32_t i = 0; i < SBGL_MAX_SHADERS; i++) {
968 if (ctx->shaders[i].active) {
969 sbgl_gfx_DestroyShader(ctx, (sbgl_Shader)(i + 1));
970 }
971 }
972
973 // Clean up all active pipelines
974 for (uint32_t i = 0; i < SBGL_MAX_PIPELINES; i++) {
975 if (ctx->pipelines[i].active) {
976 sbgl_gfx_DestroyPipeline(ctx, (sbgl_Pipeline)(i + 1));
977 }
978 if (ctx->computePipelines[i].active) {
979 sbgl_gfx_DestroyComputePipeline(ctx, (sbgl_ComputePipeline)(i + 1));
980 }
981 }
982
983 // Process any remaining deferred buffers
984 for (uint32_t f = 0; f < SBGL_MAX_FRAMES_IN_FLIGHT; f++) {
985 for (uint32_t i = 0; i < ctx->deferredCount[f]; i++) {
986 sbgl_gfx_DestroyBuffer(ctx, ctx->deferredBuffers[f][i]);
987 }
988 ctx->deferredCount[f] = 0;
989 }
990
991 for (uint32_t i = 0; i < SBGL_MAX_FRAMES_IN_FLIGHT; i++) {
992 ctx->vk.vkDestroyFence(ctx->device, ctx->inFlightFences[i], NULL);
993 }
994
995 for (uint32_t i = 0; i < SBGL_MAX_SWAPCHAIN_IMAGES; i++) {
996 if (ctx->imageAvailableSemaphores[i] != VK_NULL_HANDLE) {
997 ctx->vk.vkDestroySemaphore(ctx->device, ctx->imageAvailableSemaphores[i], NULL);
998 }
999 if (ctx->renderFinishedSemaphores[i] != VK_NULL_HANDLE) {
1000 ctx->vk.vkDestroySemaphore(ctx->device, ctx->renderFinishedSemaphores[i], NULL);
1001 }
1002 }
1003 ctx->vk.vkDestroyQueryPool(ctx->device, ctx->queryPool, NULL);
1004 ctx->vk.vkDestroyCommandPool(ctx->device, ctx->commandPool, NULL);
1005
1006 ctx->vk.vkFreeMemory(ctx->device, ctx->staticHeap.memory, NULL);
1007 ctx->vk.vkFreeMemory(ctx->device, ctx->dynamicHeap.memory, NULL);
1008 ctx->vk.vkFreeMemory(ctx->device, ctx->managedHeap.memory, NULL);
1009
1010 cleanup_swapchain(ctx);
1011 ctx->vk.vkDestroyDevice(ctx->device, NULL);
1012 }
1013 if (ctx->instance) {
1014 vkDestroySurfaceKHR(ctx->instance, ctx->surface, NULL);
1015 vkDestroyInstance(ctx->instance, NULL);
1016 }
1017}

1019bool sbgl_gfx_BeginFrame(sbgl_GfxContext* ctx) {
1020 ctx->vk.vkWaitForFences(
1021 ctx->device,
1022 1,
1023 &ctx->inFlightFences[ctx->currentFrame],
1024 VK_TRUE,
1025 UINT64_MAX
1026 );
1027
1028 /* The system processes the deferred destruction queue for the current frame slot,
1029 releasing GPU resources that are no longer in flight. */
1030 for (uint32_t i = 0; i < ctx->deferredCount[ctx->currentFrame]; i++) {
1031 sbgl_gfx_DestroyBuffer(ctx, ctx->deferredBuffers[ctx->currentFrame][i]);
1032 }
1033 ctx->deferredCount[ctx->currentFrame] = 0;
1034
1035 /* The transient allocation offset is reset for the current frame, effectively
1036 recycling the GPU memory for new data while ensuring it does not overlap with
1037 memory currently in use by other frames in flight. */
1038 ctx->transientOffsets[ctx->currentFrame] = 0;
1039 ctx->dynamicHeap.offset[ctx->currentFrame] = 0;
1040
1041 if (sbgl_os_WasWindowResized(ctx->window)) {
1042 recreate_swapchain(ctx);
1043 }
1044
1045 VkResult result = ctx->vk.vkAcquireNextImageKHR(
1046 ctx->device,
1047 ctx->swapchain,
1048 UINT64_MAX,
1049 ctx->imageAvailableSemaphores[ctx->semaphoreIndex],
1050 VK_NULL_HANDLE,
1051 &ctx->currentImageIndex
1052 );
1053
1054 if (result == VK_ERROR_OUT_OF_DATE_KHR) {
1055 recreate_swapchain(ctx);
1056 return false;
1057 } else if (result != VK_SUCCESS && result != VK_SUBOPTIMAL_KHR) {
1058 return false;
1059 }
1060
1061 ctx->vk.vkResetFences(ctx->device, 1, &ctx->inFlightFences[ctx->currentFrame]);
1062 ctx->vk.vkResetCommandBuffer(ctx->commandBuffers[ctx->currentFrame], 0);
1063 VkCommandBufferBeginInfo beginInfo = { .sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_BEGIN_INFO };
1064 ctx->vk.vkBeginCommandBuffer(ctx->commandBuffers[ctx->currentFrame], &beginInfo);
1065
1066 /* The system resets the query pool for the current frame to prepare for new
1067 timestamp recordings. */
1068 ctx->vk.vkCmdResetQueryPool(
1069 ctx->commandBuffers[ctx->currentFrame],
1070 ctx->queryPool,
1071 ctx->currentFrame * 2,
1072 2
1073 );
1074
1075 return true;
1076}

1188void sbgl_gfx_EndFrame(sbgl_GfxContext* ctx) {
1189 ctx->vk.vkEndCommandBuffer(ctx->commandBuffers[ctx->currentFrame]);
1190
1191 VkPipelineStageFlags waitStages[] = { VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT };
1192
1193 /* The system utilizes a semaphore indexed by the current image to signal completion
1194 to the presentation engine, preventing reuse conflicts during high-frequency updates. */
1195 VkSemaphore signalSemaphore = ctx->renderFinishedSemaphores[ctx->currentImageIndex];
1196
1197 VkSubmitInfo submitInfo = {
1198 .sType = VK_STRUCTURE_TYPE_SUBMIT_INFO,
1199 .waitSemaphoreCount = 1,
1200 .pWaitSemaphores = &ctx->imageAvailableSemaphores[ctx->semaphoreIndex],
1201 .pWaitDstStageMask = waitStages,
1202 .commandBufferCount = 1,
1203 .pCommandBuffers = &ctx->commandBuffers[ctx->currentFrame],
1204 .signalSemaphoreCount = 1,
1205 .pSignalSemaphores = &signalSemaphore,
1206 };
1207 ctx->vk

1234sbgl_Buffer
1235sbgl_gfx_CreateBuffer(sbgl_GfxContext* ctx, sbgl_BufferUsage usage, size_t size, const void* data) {
1236 /* Search for an available buffer slot in the internal tracking arrays. */
1237 uint32_t index = 0;
1238 for (; index < SBGL_MAX_BUFFERS; index++) {
1239 if (!ctx->bufferActive[index])
1240 break;
1241 }
1242 if (index == SBGL_MAX_BUFFERS)
1243 return SBGL_INVALID_HANDLE;
1244
1245 /* Identify the target memory heap based on the buffer's intended usage.
1246 Vertex and index buffers are assigned to the static heap, while storage
1247 buffers utilize the managed heap for persistence. */
1248 SBGL_HeapType heapType = SBGL_HEAP_TYPE_MANAGED;
1249 if (usage & (SBGL_BUFFER_USAGE_VERTEX | SBGL_BUFFER_USAGE_INDEX)) {
1250 heapType = SBGL_HEAP_TYPE_STATIC;
1251 } else if (usage & SBGL_BUFFER_USAGE_STORAGE) {
1252 heapType = SBGL_HEAP_TYPE_MANAGED;
1253 }
1254
1255 VkBufferCreateInfo bufferInfo = {
1256 .sType = VK_STRUCTURE_TYPE_BUFFER_CREATE_INFO,
1257 .size = size,
1258 .usage = (usage & SBGL_BUFFER_USAGE_VERTEX ? VK_BUFFER_USAGE_VERTEX_BUFFER_BIT : 0) |
1259 (usage & SBGL_BUFFER_USAGE_INDEX ? VK_BUFFER_USAGE_INDEX_BUFFER_BIT : 0) |
1260 (usage & SBGL_BUFFER_USAGE_STORAGE ? VK_BUFFER_USAGE_STORAGE_BUFFER_BIT : 0) |
1261 (usage & SBGL_BUFFER_USAGE_INDIRECT ? VK_BUFFER_USAGE_INDIRECT_BUFFER_BIT : 0) |
1262 (usage & SBGL_BUFFER_USAGE_TRANSFER_DST ? VK_BUFFER_USAGE_TRANSFER_DST_BIT : 0) |
1263 VK_BUFFER_USAGE_SHADER_DEVICE_ADDRESS_BIT,
1264 .sharingMode = VK_SHARING_MODE_EXCLUSIVE,
1265 };
1266
1267 SBGL_VulkanBuffer* buffer = &ctx->buffers[index];
1268 if (ctx->vk.vkCreateBuffer(ctx->device, &bufferInfo, NULL, &buffer->handle) != VK_SUCCESS) {
1269 return SBGL_INVALID_HANDLE;
1270 }
1271
1272 VkMemoryRequirements memRequirements;
1273 ctx->vk.vkGetBufferMemoryRequirements(ctx->device, buffer->handle, &memRequirements);
1274
1275 /* Sub-allocate the required memory range from the selected hybrid heap. */
1276 uint32_t offset = SBGL_INVALID_OFFSET;
1277 VkDeviceMemory heapMemory = VK_NULL_HANDLE;
1278 void* heapMappedBase = NULL;
1279
1280 switch (heapType) {
1281 case SBGL_HEAP_TYPE_STATIC:
1282 offset = static_heap_alloc(ctx, memRequirements.size);
1283 heapMemory = ctx->staticHeap.memory;
1284 heapMappedBase = ctx->staticHeap.mapped;
1285 break;
1286 case SBGL_HEAP_TYPE_DYNAMIC:
1287 offset = dynamic_heap_alloc(ctx, memRequirements.size);
1288 heapMemory = ctx->dynamicHeap.memory;
1289 heapMappedBase = ctx->dynamicHeap.mapped[ctx->currentFrame];
1290 break;
1291 case SBGL_HEAP_TYPE_MANAGED:
1292 offset = managed_heap_alloc(ctx, memRequirements.size);
1293 heapMemory = ctx->managedHeap.memory;
1294 heapMappedBase = ctx->managedHeap.mapped;
1295 break;
1296 }
1297
1298 if (offset == SBGL_INVALID_OFFSET) {
1299 ctx->vk.vkDestroyBuffer(ctx->device, buffer->handle, NULL);
1300 return SBGL_INVALID_HANDLE;
1301 }
1302
1303 /* Bind the buffer handle to the sub-allocated memory region within the heap. */
1304 ctx->vk.vkBindBufferMemory(ctx->device, buffer->handle, heapMemory, offset);
1305
1306 buffer->size = size;
1307 buffer->offset = offset;
1308 buffer->heapType = heapType;
1309 buffer->mapped = (char*)heapMappedBase + offset;
1310 ctx->bufferActive[index] = true;
1311
1312 /* If initial data is provided, perform an immediate memory copy to the
1313 persistently mapped buffer address. */
1314 if (data && buffer->mapped) {
1315 memcpy(buffer->mapped, data, size);
1316 }
1317
1318 return (sbgl_Buffer)(index + 1);
1319}

1321void sbgl_gfx_DestroyBuffer(sbgl_GfxContext* ctx, sbgl_Buffer handle) {
1322 /* The system releases the GPU-side buffer handle and, if the memory was
1323 allocated from the managed heap, returns the range to the sub-allocator
1324 to mitigate memory fragmentation. Static and Dynamic allocations are
1325 reclaimed automatically or persist until shutdown. */
1326 if (handle == SBGL_INVALID_HANDLE)
1327 return;
1328 uint32_t index = (uint32_t)handle - 1;
1329 if (index >= SBGL_MAX_BUFFERS || !ctx->bufferActive[index])
1330 return;
1331
1332 SBGL_VulkanBuffer* buffer = &ctx->buffers[index];
1333 ctx->vk.vkDestroyBuffer(ctx->device, buffer->handle, NULL);
1334
1335 if (buffer->heapType == SBGL_HEAP_TYPE_MANAGED) {
1336 managed_heap_free(ctx, buffer->offset);
1337 }
1338
1339 ctx->bufferActive[index] = false;
1340}

1781void sbgl_gfx_MemoryBarrier(sbgl_GfxContext* ctx, sbgl_BarrierType type) {
1782 /* The system injects a pipeline barrier into the command stream to synchronize
1783 memory access between different execution stages, preventing race conditions. */
1784 VkMemoryBarrier barrier = { .sType = VK_STRUCTURE_TYPE_MEMORY_BARRIER };
1785 VkPipelineStageFlags srcStage = 0;
1786 VkPipelineStageFlags dstStage = 0;
1787
1788 switch (type) {
1789 case SBGL_BARRIER_COMPUTE_TO_COMPUTE:
1790 /* Synchronizes compute and transfer (fill) writes to be visible to
1791 subsequent compute operations. */
1792 barrier.srcAccessMask = VK_ACCESS_SHADER_WRITE_BIT | VK_ACCESS_TRANSFER_WRITE_BIT;
1793 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT | VK_ACCESS_SHADER_WRITE_BIT;
1794 srcStage = VK_PIPELINE_STAGE_COMPUTE_SHADER_BIT | VK_PIPELINE_STAGE_TRANSFER_BIT;
1795 dstStage = VK_PIPELINE_STAGE_COMPUTE_SHADER_BIT;
1796 break;
1797 case SBGL_BARRIER_COMPUTE_TO_INDIRECT:
1798 /* Synchronizes compute writes to SSBOs for use in indirect draw command buffers. */
1799 barrier.srcAccessMask = VK_ACCESS_SHADER_WRITE_BIT;
1800 barrier.dstAccessMask = VK_ACCESS_INDIRECT_COMMAND_READ_BIT;
1801 srcStage = VK_PIPELINE_STAGE_COMPUTE_SHADER_BIT;
1802 dstStage = VK_PIPELINE_STAGE_DRAW_INDIRECT_BIT;
1803 break;
1804 case SBGL_BARRIER_COMPUTE_TO_GRAPHICS:
1805 /* Synchronizes compute writes to be visible to vertex input and shader stages. */
1806 barrier.srcAccessMask = VK_ACCESS_SHADER_WRITE_BIT;
1807 barrier.dstAccessMask = VK_ACCESS_VERTEX_ATTRIBUTE_READ_BIT | VK_ACCESS_SHADER_READ_BIT;
1808 srcStage = VK_PIPELINE_STAGE_COMPUTE_SHADER_BIT;
1809 dstStage = VK_PIPELINE_STAGE_VERTEX_INPUT_BIT | VK_PIPELINE_STAGE_VERTEX_SHADER_BIT;
1810 break;
1811 case SBGL_BARRIER_GRAPHICS_TO_COMPUTE:
1812 /* Synchronizes graphics writes to be visible to subsequent compute operations. */
1813 barrier.srcAccessMask = VK_ACCESS_SHADER_WRITE_BIT;
1814 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT | VK_ACCESS_SHADER_WRITE_BIT;
1815 srcStage = VK_PIPELINE_STAGE_VERTEX_SHADER_BIT | VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT;
1816 dstStage = VK_PIPELINE_STAGE_COMPUTE_SHADER_BIT;
1817 break;
1818 case SBGL_BARRIER_HOST_TO_COMPUTE:
1819 /* Synchronizes host writes to be visible to subsequent compute operations. */
1820 barrier.srcAccessMask = VK_ACCESS_HOST_WRITE_BIT;
1821 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT | VK_ACCESS_SHADER_WRITE_BIT;
1822 srcStage = VK_PIPELINE_STAGE_HOST_BIT;
1823 dstStage = VK_PIPELINE_STAGE_COMPUTE_SHADER_BIT;
1824 break;
1825 case SBGL_BARRIER_HOST_TO_GRAPHICS:
1826 /* Synchronizes host writes to be visible to subsequent graphics (vertex) operations. */
1827 barrier.srcAccessMask = VK_ACCESS_HOST_WRITE_BIT;
1828 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
1829 srcStage = VK_PIPELINE_STAGE_HOST_BIT;
1830 dstStage = VK_PIPELINE_STAGE_VERTEX_SHADER_BIT;
1831 break;
1832 }
1833
1834 ctx->vk.vkCmdPipelineBarrier(
1835 ctx->commandBuffers[ctx->currentFrame],
1836 srcStage,
1837 dstStage,
1838 0,
1839 1,
1840 &barrier,
1841 0,
1842 NULL,
1843 0,
1844 NULL
1845 );
1846}
